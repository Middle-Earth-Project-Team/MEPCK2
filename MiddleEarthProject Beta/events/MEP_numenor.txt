##############  TEMPESTS EVENTS ##############
###### Happens if n army spawn near to Meneltarma province ######


### Tempest ! ###
narrative_event = {
	id = numenor.5
	title = "EVTNAMEnumenor.5"
	desc = "EVTDESCnumenor.5"
	picture = GFX_evt_gondor_kingdom
	
	trigger = {
		OR = {
			any_playable_ruler = {
				in_command = yes
				location  = {
					OR = {
						province_id = 795
						province_id = 796
					}
				}
				count = 1
			}
			any_playable_ruler = {
				any_courtier = {
					in_command = yes
					location  = {
						OR = {
							province_id = 795
							province_id = 796
						}
					}
				}
				count = 1
			}			
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	immediate = {
		set_character_flag = in_tempest_meneltarma
	}

	option = {
		name = "EVTOPTAnumenor.5" ## I'll try to see this land !
		random_list = {
			75 = {
			}
			25 = {
				add_trait = brave
			}
		}
		if = { 
			limit = { ## The King of Gondor or Arnor, direct heir of Elendil.
				OR = {
					has_landed_title = e_arnor
					has_landed_title = e_gondor
				}
				religion = religion_dunedain			
				OR = {
					trait = blood_of_isildur
					trait = blood_of_anarion
				}	
				OR = {
					trait = narsil
					trait = anduril
				}
				trait = barahir
			}
			narrative_event = { id = numenor.6 days = 2 } ## He'll survive
		}	
		if = { 
			limit = { ## The King of Gondor or Arnor, not direct heir of Elendil.
				OR = {
					has_landed_title = e_arnor
					has_landed_title = e_gondor
				}
				religion = religion_dunedain			
				OR = {
					trait = blood_of_isildur
					trait = blood_of_anarion
				}	
				OR = {
					NOT = { trait = narsil }
					NOT = { trait = anduril }
				}
				NOT = { trait = barahir }
			}
			random_list = {
				70 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				30 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}
		if = { 
			limit = { ## Isn't a dunedain, an ainur or an elf - good side
				NOT = { religion = religion_dunedain }		
				NOT = { religion_group = religion_group_eldar}
				NOT = { culture = culture_maiar }
				NOT = { culture = culture_valar }
				NOT = { culture = culture_istari }
				NOT = { 
					trait = evil_side
				}
			}
			random_list = {
				15 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				85 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}
		if = { 
			limit = { ## Is an ainur
				OR = {
					culture = culture_maiar
					culture = culture_valar
					culture = culture_istari
				}
			}
			random_list = {
				80 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				20 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}	
		if = { 
			limit = { ## Is an elf - good side
				religion_group = religion_group_eldar
				NOT = { 
					trait = evil_side
				}
			}
			random_list = {
				60 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				40 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}	
		if = { 
			limit = { ## Is an elf - evil side
				religion_group = religion_group_eldar
				trait = evil_side
			}
			random_list = {
				10 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				90 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}
		if = { 
			limit = { ## Isn't a dunedain, an ainur or an elf - evil side
				NOT = { religion = religion_dunedain }		
				NOT = { religion_group = religion_group_eldar }
				NOT = { culture = culture_maiar }
				NOT = { culture = culture_valar }
				NOT = { culture = culture_istari }
				trait = evil_side
			}
			random_list = {
				1 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				99 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}
		if = { 
			limit = { ## Is a dunedain - good side
				religion = religion_dunedain
				NOT = { 
					trait = evil_side
				}
			}
			random_list = {
				25 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				75 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}	
		if = { 
			limit = { ## Is a dunedain - evil side
				religion = religion_dunedain
				trait = evil_side
			}
			random_list = {
				2 = {				
					narrative_event = { id = numenor.9 days = 2 } ## He'll survive
				}
				98 = {
					narrative_event = { id = numenor.7 days = 2 } ## He'll die
				}
			}
		}		
	}
	
	option = {
		name = "EVTOPTAnumenor.5" ## I'll return to Middle-Earth.
		prestige = -50
		ROOT = {
			troops = -0.6
			morale = -0.6
		}
		remove_claim = c_meneltarma
	}
}
